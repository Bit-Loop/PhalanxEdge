# Proxmox Management Cron Jobs
# Add these lines to your crontab with 'crontab -e'

# Update Proxmox inventory daily at 6:00 AM
0 6 * * * $HOME/.ansible/scripts/proxmox-manager.sh update --quiet > /dev/null 2>&1

# Run monitoring and generate reports on Monday, Wednesday, Friday at 8:00 AM
0 8 * * 1,3,5 $HOME/.ansible/scripts/proxmox-manager.sh monitor --quiet > /dev/null 2>&1

# Check SSH key access weekly on Sunday at 7:00 AM
0 7 * * 0 $HOME/.ansible/scripts/setup_ssh_keys.sh > /dev/null 2>&1

# Full inventory backup on the 1st of each month at 2:00 AM
0 2 1 * * $HOME/.ansible/scripts/proxmox-manager.sh backup --quiet > /dev/null 2>&1
